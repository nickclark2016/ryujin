171  {
172  // Test for zero length strings, in which case we have a match or a failure,
173  // but the return value is the same either way.
174  if ((p1Begin == p1End) || (p2Begin == p2End))
175  return p1Begin;
176 
177  // Test for a pattern of length 1.
178  if ((p2Begin + 1) == p2End)
179  return CharTypeStringFindEnd(p1Begin, p1End, *p2Begin);
180 
181  // Test for search string length being longer than string length.
182  if ((p2End - p2Begin) > (p1End - p1Begin))
183  return p1End;
184 
185  // General case.
186  const typename basic_string<Type, Allocator>::value_type* pSearchEnd = (p1End - (p2End - p2Begin) + 1);
187  const typename basic_string<Type, Allocator>::value_type* pCurrent1;
188  const typename basic_string<Type, Allocator>::value_type* pCurrent2;
189 
190  while (pSearchEnd != p1Begin)
191  {
192  // Search for the last occurrence of *p2Begin.
193  pCurrent1 = CharTypeStringFindEnd(p1Begin, pSearchEnd, *p2Begin);
194  if (pCurrent1 == pSearchEnd) // If the first char of p2 wasn't found,
195  return p1End; // then we immediately have failure.
196 
197  // In this case, *pTemp == *p2Begin. So compare the rest.
198  pCurrent2 = p2Begin;
199  while (*pCurrent1++ == *pCurrent2++)
200  {
201  if (pCurrent2 == p2End)
202  return (pCurrent1 - (p2End - p2Begin));
203  }
204
