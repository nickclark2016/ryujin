57  {
58  quat<T> quat(as<T>(0));
59 
60  // Compute the trace to see if it is positive or not.
61  const T trace = m[0][0] + m[1][1] + m[2][2];
62 
63  // check the sign of the trace
64  if (trace > 0) {
65  // trace is positive
66  T s = std::sqrt(trace + 1);
67  quat.w = T(0.5) * s;
68  s = T(0.5) / s;
69  quat.x = (m[1][2] - m[2][1]) * s;
70  quat.y = (m[2][0] - m[0][2]) * s;
71  quat.z = (m[0][1] - m[1][0]) * s;
72  }
73  else {
74  // trace is negative
75 
76  // Find the index of the greatest diagonal
77  size_t i = 0;
78  if (m[1][1] > m[0][0]) { i = 1; }
79  if (m[2][2] > m[i][i]) { i = 2; }
80

82  constexpr size_t next_ijk[3] = { 1, 2, 0 };
83  size_t j = next_ijk[i];
84  size_t k = next_ijk[j];
85  T s = std::sqrt((m[i][i] - (m[j][j] + m[k][k])) + 1);
86  quat[i] = T(0.5) * s;
87  if (s != 0) {
88  s = T(0.5) / s;
89  }
90  quat.w = (m[j][k] - m[k][j]) * s;
91  quat[j] = (m[i][j] + m[j][i]) * s;
92  quat[k] = (m[i][k] + m[k][i]) * s;

260  scale.y = norm(row[1]);
261  scale.z = norm(row[2]);

264  if (trace > as<T>(0))

269  rotation.x = root * (row[1].z - row[2].y);
270  rotation.y = root * (row[2].x - row[0].z);
271  rotation.z = root * (row[0].y - row[1].x);
272  }

276  sz i = 0;

294  inline constexpr mat4<T> perspective(const T aspect, const T fov, const T near, const T far)
295  {
296  assert(near < far && "Near plane distance must be less than far plane distance");
297  // 1 / (aspect * tan(fov / 2)) 0 0 0
298  // 0 1 / tan(fov / 2) 0 0
299  // 0 0 (-near - far) / (near - far) (2 * near * far) / (near - far)
300  // 0 0 1 0
301  const T fovRad = as_radians(fov);
302  const T tanFov2 = as<T>(std::tan(fovRad / as<T>(2)));
303  const T invTanFov2 = as<T>(1) / tanFov2;
304  const T invAspectTanFov2 = invTanFov2 * (as<T>(1) / aspect);
305  const T nearMinusFar = near - far;
306 
307  const T zero = as<T>(0);
308 
309  const vec4<T> col0(invAspectTanFov2, zero, zero, zero);
310  const vec4<T> col1(zero, invTanFov2, zero, zero);
311  const vec4<T> col2(zero, zero, (-near - far) / nearMinusFar, as<T>(1));
312  const vec4<T> col3(zero, zero, (as<T>(2) * near * far) / nearMinusFar, zero);
313  return mat4(col0, col1, col2, col3);
314  }
